// Write a C++ program to read the content of file “JOBS.TXT” and display
// the contents on the screen.

#include <iostream>
#include <fstream>
using namespace std;

int main()
{
    ifstream inFile;
    inFile.open("JOBS.TXT");
    string line;
    while (getline(inFile, line))
    {
        cout << line << endl;
    }
    inFile.close();
    return 0;
}
